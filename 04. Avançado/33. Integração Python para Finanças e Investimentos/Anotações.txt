        data_fim = datetime.now().strftime('%Y-%m-%d')

    Contruindo um função para obter dados de varias ações:
        def pegar_cotacoes(lista_tickers, data_inicio, data_fim):
            df = yf.download(lista_tickers, data_inicio, data_fim)
            df = df.ffill()
            return df
